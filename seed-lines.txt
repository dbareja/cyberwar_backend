186 const seed = async () => {
187   try {
188     await connectDB();
189     console.log('🌱 Starting database seeding...');
190 
191     // Clear existing data
192     await Promise.all([
193       User.deleteMany({}),
194       Category.deleteMany({}),
195       Tag.deleteMany({}),
196       Blog.deleteMany({})
197     ]);
198     console.log('✅ Cleared existing data');
199 
200     // Create admin user
201     const adminUser = await User.create({
202       name: 'CyberWar Admin',
203       email: process.env.ADMIN_EMAIL || 'admin@cyberwar.com',
204       password: process.env.ADMIN_PASSWORD || 'Admin@123456',
205       role: 'admin',
206       bio: 'Chief Security Editor at CyberWar Intelligence'
207     });
208     console.log('✅ Admin user created:', adminUser.email);
209 
210     // Create editor
211     const editor = await User.create({
212       name: 'Sarah Mitchell',
213       email: 'sarah@cyberwar.com',
214       password: 'Editor@123456',
215       role: 'editor',
216       bio: 'Senior Cybersecurity Analyst and Editor'
217     });
218 
219     // Create categories
220     const createdCategories = await Category.insertMany(categories);
221     console.log(`✅ Created ${createdCategories.length} categories`);
222 
223     // Create tags
224     const createdTags = await Tag.insertMany(tags);
225     console.log(`✅ Created ${createdTags.length} tags`);
226 
227     // Create blogs
228     const categoryMap = {};
229     createdCategories.forEach(c => categoryMap[c.slug] = c._id);
230 
231     const blogCategoryAssignments = ['cybersecurity', 'vulnerabilities', 'ransomware', 'cybersecurity', 'cybersecurity', 'vulnerabilities', 'nation-state', 'vulnerabilities'];
232     const enrichedBlogs = sampleBlogs.map((blog, index) => ({
233       ...blog,
234       author: index % 2 === 0 ? adminUser._id : editor._id,
235       category: categoryMap[blogCategoryAssignments[index]] || createdCategories[0]._id,
236       tags: [createdTags[index % createdTags.length]._id, createdTags[(index + 1) % createdTags.length]._id],
237       publishedAt: new Date(Date.now() - (index * 24 * 60 * 60 * 1000))
238     }));
239 
240     // Create blogs one by one to trigger slug generation
241     const createdBlogs = []
242     for (const blogData of enrichedBlogs) {
243       const blog = await Blog.create(blogData)
244       createdBlogs.push(blog)
245     }
246     console.log(`✅ Created ${createdBlogs.length} blog posts`);
247 
248     console.log('\n🎉 Database seeded successfully!');
249     console.log('\n📋 Login Credentials:');
250     console.log('   Admin Email:', adminUser.email);
251     console.log('   Admin Password:', process.env.ADMIN_PASSWORD || 'Admin@123456');
252     console.log('\n🌐 API Endpoint: http://localhost:5000/api');
253 
254     process.exit(0);
255   } catch (error) {
256     console.error('❌ Seeding error:', error);
257     process.exit(1);
258   }
259 };
